\newcommand*\keyword[2][]{%
	\edef\rpTmpKw{\ifx&#1&#2\else#1\fi}% http://www.latex-community.org/forum/viewtopic.php?f=5&t=5976
	\ifx\rpKeywordList\empty%
		\edef\rpKeywordList{\rpTmpKw}%
	\else%
		\edef\rpKeywordList{\rpKeywordList, \rpTmpKw}%
	\fi%
	\textbf{#2}%
}

% \rpResumeAuto{ -- français -- }{ -- anglais -- }
\long\def\rpResumeAuto#1#2{\bgroup
	\sectionSpeciale{Résumé}
	\def\rpKeywordList{}
	#1
	
	\ifx\rpKeywordList\empty\else
		\rpKeywords{Mots-clés :}{\rpKeywordList}
	\fi
	
	\vfill
	
	% Oublions un instant la typographie française
	\begin{otherlanguage*}{english}
		\sectionSpeciale{Abstract}
		\def\rpKeywordList{}
		#2
		
		\ifx\rpKeywordList\empty\else
			\rpKeywords{Keywords:}{\rpKeywordList}
		\fi
	\end{otherlanguage*}
	
	\vfill
\egroup}
